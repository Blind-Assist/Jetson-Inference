UID_NUM=$(id -u blind)

User=blind
Group=blind

Environment=PYTHONUNBUFFERED=1
Environment=XDG_RUNTIME_DIR=/run/user/$UID_NUM

ExecStart=/home/blind/internvl3/myenv/bin/python -u /home/blind/internvl3/InternVL/internvl_chat/webcam_audio.py
